## Virtual Environments

### `mkvenv` and `avenv` shell helpers

Appends two functions to `.bashrc`: `mkvenv` creates a named venv based on the current directory name, and `avenv` activates it. This avoids name collisions when managing many project venvs.

```bash
#!/usr/bin/env bash

function mkvenv() {
    local curdir
    curdir=$(basename "$(pwd)")
    mkdir -p ~/.virtualenvs
    python3 -m venv ~/.virtualenvs/"${curdir}"
    echo "Virtual environment for ${curdir} created"
}

function avenv() {
    local curdir
    curdir=$(basename "$(pwd)")
    if [ -d ~/.virtualenvs/"${curdir}" ]; then
        source ~/.virtualenvs/"${curdir}"/bin/activate
        echo "Virtual environment for ${curdir} activated"
    else
        echo "Error: No virtual environment found for ${curdir}"
    fi
}

echo "" >> ~/.bashrc
echo "# Define mkvenv function" >> ~/.bashrc
declare -f mkvenv >> ~/.bashrc
echo "" >> ~/.bashrc
echo "# Define avenv function" >> ~/.bashrc
declare -f avenv >> ~/.bashrc
```

## Azure Authentication

### Authenticate with a Service Principal via environment variables

```python
import os
from azure.identity import ClientSecretCredential

def azure_authenticate():
    credentials = ClientSecretCredential(
        client_id=os.environ["ARM_CLIENT_ID"],
        client_secret=os.environ["ARM_CLIENT_SECRET"],
        tenant_id=os.environ["ARM_TENANT_ID"],
    )
    return credentials
```

### List resources in a resource group using the Resource Management client

```python
import os
from azure.identity import ClientSecretCredential
from azure.mgmt.resource import ResourceManagementClient


def azure_authenticate():
    return ClientSecretCredential(
        client_id=os.environ["ARM_CLIENT_ID"],
        client_secret=os.environ["ARM_CLIENT_SECRET"],
        tenant_id=os.environ["ARM_TENANT_ID"],
    )


class AzureResourceChecker:
    @staticmethod
    def check_resources(resource_group: str, resource_type: str):
        subscription_id = os.environ["ARM_SUBSCRIPTION_ID"]
        client = ResourceManagementClient(azure_authenticate(), subscription_id)

        resources = [
            r for r in client.resources.list_by_resource_group(resource_group)
            if resource_type in r.type
        ]

        if resources:
            print(f"{resource_group} contains {resource_type}:")
            for r in resources:
                print(f"  - {r.name}")
        else:
            print(f"{resource_group} has no {resource_type} resources.")


obj = AzureResourceChecker()
obj.check_resources("rg-ldo-euw-dev-mgt", "Microsoft.KeyVault/vaults")
```

### Query the Azure Instance Metadata Service from a VM

Retrieves compute metadata from inside a VM without any credentials — useful for dynamic configuration in user-data scripts or agents.

```python
import json
import os
import requests

os.environ["NO_PROXY"] = "*"   # bypass any proxy for link-local addresses

url     = "http://169.254.169.254/metadata/instance?api-version=2021-02-01"
headers = {"Metadata": "true"}

r    = requests.get(url=url, headers=headers)
data = r.json()

print(json.dumps(data, indent=2))

subscription_id = data["compute"]["subscriptionId"]
resource_id     = data["compute"]["resourceId"]
segments        = resource_id.split("/")

print(f"Subscription:   {segments[2]}")
print(f"Resource group: {segments[4]}")
print(f"Provider:       {segments[6]}")
print(f"Resource type:  {segments[7]}")
print(f"Resource name:  {segments[8]}")
```

### Azure Function with user-assigned Managed Identity

```python
import azure.functions as func
import logging
import datetime
import os
from azure.identity import DefaultAzureCredential
from azure.mgmt.resource import ResourceManagementClient

subscription_id     = os.environ["ARM_SUBSCRIPTION_ID"]
client_id           = os.environ["ARM_CLIENT_ID"]
resource_group_name = os.environ["RESOURCE_GROUP_NAME"]


def main(timer: func.TimerRequest) -> None:
    ts         = datetime.datetime.utcnow().isoformat()
    credential = DefaultAzureCredential(managed_identity_client_id=client_id)
    client     = ResourceManagementClient(credential, subscription_id)

    for resource in client.resources.list_by_resource_group(resource_group_name):
        logging.info("Resource: %s", resource.id)

    logging.info("Timer trigger ran at %s", ts)
```

## Utilities

### Coloured logger

```python
import logging


class ColouredFormatter(logging.Formatter):
    CYAN   = "\033[1;36m"
    RED    = "\033[1;31m"
    YELLOW = "\033[1;33m"
    PURPLE = "\033[0;35m"
    RESET  = "\033[0m"
    FMT    = "%(asctime)s - %(name)s - %(levelname)s - %(message)s"

    FORMATS = {
        logging.DEBUG:    CYAN   + FMT + RESET,
        logging.INFO:     PURPLE + FMT + RESET,
        logging.WARNING:  YELLOW + FMT + RESET,
        logging.ERROR:    RED    + FMT + RESET,
        logging.CRITICAL: RED    + FMT + RESET,
    }

    def format(self, record):
        return logging.Formatter(self.FORMATS[record.levelno]).format(record)


logger = logging.getLogger(__name__)
logger.setLevel(logging.DEBUG)
ch = logging.StreamHandler()
ch.setFormatter(ColouredFormatter())
logger.addHandler(ch)
```

## Terraform Utilities

### Sort `variables.tf` alphabetically

```python
import re
import argparse


def read_file(path: str) -> str:
    with open(path) as f:
        return f.read()


def write_file(path: str, content: str) -> None:
    with open(path, "w") as f:
        f.write(content)


def sort_variables(content: str) -> str:
    pattern = re.compile(r'variable\s+"[^"]+"\s+\{.*?\n}', re.DOTALL)
    blocks  = pattern.findall(content)
    blocks.sort(key=lambda b: re.search(r'variable\s+"([^"]+)"', b).group(1))
    return "\n\n".join(blocks)


def main():
    parser = argparse.ArgumentParser(description="Sort Terraform variables alphabetically.")
    parser.add_argument("-i", "--in-file",  default="./variables.tf")
    parser.add_argument("-o", "--out-file", default="./variables.tf")
    args = parser.parse_args()

    write_file(args.out_file, sort_variables(read_file(args.in_file)))
    print(f"Sorted variables written to {args.out_file}")


if __name__ == "__main__":
    main()
```

### Sort `outputs.tf` alphabetically

```python
import re
import argparse


def sort_outputs(content: str) -> str:
    pattern = re.compile(r'output\s+"[^"]+"\s+\{.*?\n}', re.DOTALL)
    blocks  = pattern.findall(content)
    blocks.sort(key=lambda b: re.search(r'output\s+"([^"]+)"', b).group(1))
    return "\n\n".join(blocks)


def main():
    parser = argparse.ArgumentParser(description="Sort Terraform outputs alphabetically.")
    parser.add_argument("-i", "--in-file",  default="./outputs.tf")
    parser.add_argument("-o", "--out-file", default="./outputs.tf")
    args = parser.parse_args()

    with open(args.in_file)  as f: content = f.read()
    with open(args.out_file, "w") as f: f.write(sort_outputs(content))
    print(f"Sorted outputs written to {args.out_file}")


if __name__ == "__main__":
    main()
```
